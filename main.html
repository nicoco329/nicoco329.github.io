Assignment 1

1 What is Java programming language? Discuss its key features and advantages. ▼
Answer

Java is a high-level, general purpose, object oriented programming language. It is owned by Oracle and used for mobile applications, desktop applications, games, and many more. One of its advantages is that it works on many different platforms such as Windows, Mac, Linux, Raspberry Pi, etc. It also has a high and large demand in the current job market.

2 Explain the concept of platform independence in Java. How does Java achieve platform independence? ▼
Answer

Platform-independent means that Java's byte code can run all operating systems. Java's way of doing this is running the compiled version of Java code into a JVM.

3 Discuss the main components of a Java program and their roles ▼
Answer

Class - blueprint of a Java program.
Main Method - entry-point of a Java program.
Methods - block of code that performs a specific task.
Variables - used to store data.
Statements - instructions executed by the program.
Object - instances of class.
Comment - explanation and documentation.

4 What are the differences between JDK, JRE, and JVM in Java ▼
Answer

JDK (Java Development Kit) - used to build Java applications.
JRE (Java Runtime Environment) - provides environment to run Java programs.
JVM (Java Virtual Machine) - core execution engine of Java.

5 Explain the difference between interpreted and compiled languages. Is Java an interpreted or compiled language? ▼
Answer

Compiled Languages - converted directly into machine code that a processor can execute, leading to faster and efficient execution
Interpreted Languages - runs through a program line by line and execute each command.
Java is both compiled and interpreted

Seatwork 1 — Smart Wallet System

package Java;
import java.util.Scanner;

public class SmartWalletSystemORGAS {

	public static void main(String[] args) {
		int ch = 0, bal = 5000, cashIn = 0, trans = 0, count = 1;
		
		Scanner input = new Scanner(System.in);
		
		while(ch != 4) {
			System.out.println("===== SMART WALLET MENU =====");
			System.out.println("1. View Balance");
			System.out.println("2. Cash In");
			System.out.println("3. Send Money");
			System.out.println("4. Exit");
			System.out.println("Transaction Count: " + count);
			System.out.print("Choice: ");
			
			ch = input.nextInt();
			
			switch(ch) {
				case 1:
					System.out.println("Your current balance is: " + bal);
					count ++;
				break;
				
				case 2:
					System.out.println("Current balance: " + bal);
					System.out.print("Type amount to cash in: ");
						cashIn = input.nextInt();
						bal += cashIn;
					System.out.println("New balance: " + bal);
						cashIn = 0;
					count++;
					break;
				case 3:
					System.out.println("Current balance: " + bal);
				while (true) {
					System.out.print("Type amount to transfer (Minimum of 100, Maximum of 1500): ");
					trans = input.nextInt();
					
					if(trans < 100) {
						System.out.println("Amount is less than 100!");
					}else if(trans > 1500) {
						System.out.println("Amount is greater than 1500!");
					}else if(trans > bal) {
						System.out.println("Amount exceeds current balance!");
					}else {
						bal -= trans;
						System.out.println("Transfer successful!");
						System.out.println("New balance: " + bal);
						break;
					}
							}
						count++;
					break;
					case 4:
						count = 1;
						System.out.println("Session Terminated. Thank you!");
					break;
					
					default:
	                    System.out.println("Invalid option. Please try again.");
	                    break;
				} 
					
			}
			
			
		}
		
	}

package Java;
import java.util.Scanner;

public class StudentAgeAnalyzerOrgas {
	static void ageChecker(int x, String y){
		if (x >= 0 && x <= 12) {
			System.out.print(y + ", you are a Child.");	
		}else if(x >= 13 && x <= 19) {
			System.out.print(y + ", you are a Teenager.");	
		}else if(x >= 20 && x<=59) {
			System.out.print(y + ", you are an Adult.");	
		}else if(x >= 60) {
			System.out.print(y + ", you are a Senior Citizen.");	
		}
	}
	public static void main(String[] args) {
		Scanner userin = new Scanner(System.in);
		
			System.out.print("What is your name?: ");
		String name = userin.nextLine();
			System.out.print("How old are you?: ");
		int age = userin.nextInt();
		
		ageChecker(age, name);
		
	}

}

package Java;

public class Assignment {
	public static void main (String[] args) {
		double x = 7;
		double y = 6;
		
		x+=y;
		System.out.println(x); // x = x + y = 13 | x = 13
		
		y-=x;
		System.out.println(y); // y = y(6) - x (13)  = -7 | y = -7 		
		
		x*=y;
		System.out.println(x); // x = x (13) * y (-7) = -91 | x = -91
		
		y/=x;

                | y = 0.07692307692307693 */
		
		
	}
}

Multiple Operators

package Java;

public class Multiple_Operators {

	public static void main(String[] args) {
		int first = 12;
		int second = 8;
		
		System.out.println(first + second);
		System.out.println("first == second = " + (first == second));
		System.out.println(first > 10 && second < first);
		System.out.println(++first * second++);
		System.out.println(first+=first);
		System.out.println(first++ | ++second);

	}

}

package Java;

import java.util.Scanner;

public class EnrollmentAssessmentSystemOrgas {

	public static void main(String[] args) {
		
		String adResult ="", recProg = ""; 
		
		
		Scanner userin = new Scanner(System.in);
		
		System.out.print("Student Name: ");
		String studName = userin.nextLine();
		
		System.out.print("Senior High School Strand (STEM, ABM, HUMSS, TVL, Others): ");
		String strand = userin.nextLine();
		
		System.out.print("Entrance Exam Score (0-100): ");
		double entraScore = userin.nextDouble();
		
		System.out.print("Interview Score (0–100): ");
		double interScore = userin.nextDouble();
		
		userin.nextLine();
		
		System.out.print("First Screening Result (Pass/Fail): ");
		String screenRes = userin.nextLine();

		if("Pass".equalsIgnoreCase(screenRes)) {
			if (scoreFinal >= 85) {
				adResult = "Admitted with Scholarship";			
			}
			else if(scoreFinal >= 75) {
				adResult = "Admitted";
			}
			else if(scoreFinal >= 65) {
				adResult = "Waitlisted";
			}
			else {
				adResult = "Not Qualified";
			}
		}else {

Activity 5 — Personal Expense

package Java;
import java.util.Scanner;

public class ExpenseTrackerOrgas {

	static void titleShow() {
		System.out.println("Expense Tracker");
	}
	
	public static double ExpensesCalcu(double foo, double trans, double oth){
		
		return foo+trans+oth;
	}
	
	static String BudgetChecker(double tot, double bud) {
		if(tot > bud) {
			return "You exceeded your budget.";
		}else {
			return "You are within your budget.";
		}
		
	}
		
	
	static void Remaining(double tot, double bud, String stat) {
		double rem = bud - tot;
		
		System.out.println("Your total expenses is: " + tot);
		System.out.println("Your remaining budget is: " + rem);
		System.out.println(stat);
		
	}

		
	public static void main(String[] args) {
		
		titleShow();
		Scanner userin = new Scanner(System.in);
		
		
		System.out.print("What is your budget?");
		double budget = userin.nextInt();
		
		System.out.print("How much is your food expense?");
		double food = userin.nextInt();
		
		System.out.print("How much is your transportation expense?");
		double transport = userin.nextInt();
		
		System.out.print("How much is your other expense?");
		double other = userin.nextInt();
		
		double total = ExpensesCalcu(food, transport, other);
		String status = BudgetChecker(total,budget);
		
		Remaining(total,budget, status);
			

		}

	}
